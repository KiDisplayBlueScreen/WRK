13 Abstract:
14 
15  This module implement idle scheduling. Idle scheduling is performed when
16  the current thread is entering a wait state or must be rescheduled for
17  any other reason (e.g., affinity change), and a suitable thread cannot be
18  found after searching a suitable subset of the processor ready queues.
19 
20 --*/

32 /*++
33 
34 Routine Description:
35 
36  This function is called when the idle schedule flag is set in the current
37  PRCB. This flag is set when the current thread is entering a wait state
38  or must be rescheduled for any other reason (e.g., affinity change) and
39  a suitable thread cannot be found after searching all the processor ready
40  queues outside the dispatcher database lock. A second pass over the ready
41  queues is required since the processor was not idle during the first scan,
42  and therefore, a potential candidate thread may have been missed.
43 
44 Arguments:
45 
46  CurrentPrcb - Supplies a pointer to the current processor block.
47 
48 Return Value:
49 
50  If a thread is found to run, then a pointer to the thread is returned.
51  Otherwise, NULL is returned.
52 
53  N.B. If a thread is found, then IRQL is returned at SYNCH_LEVEL. If a
54  thread is not found, then IRQL is returned at DISPATCH_LEVEL.
55 
56 --*/
57 
58 {
59 
60  LONG Index;
61  LONG Limit;
62  LONG Number;
63  PKTHREAD NewThread;
64  ULONG Processor;
65  PKPRCB TargetPrcb;
66 
67  ASSERT (CurrentPrcb == KeGetCurrentPrcb());
68 
69  //
70  // Raise IRQL to SYNCH_LEVEL, acquire the current PRCB lock, and clear
71  // idle schedule.
72  //
73 
74  KeRaiseIrqlToSynchLevel();
75  KiAcquirePrcbLock(CurrentPrcb);
76  CurrentPrcb->IdleSchedule = FALSE;
77 
78  //
79  // If the idle thread has been selected to run on current processor, then
80  // clear the idle thread selection.
81  //
82  // N.B. This condition can occur when another thread is scheduled to run
83  // on the current processor, then becomes ineligible to run and the
84  // processor goes idle again.
85  //
86 
87  if (CurrentPrcb->NextThread == CurrentPrcb->IdleThread) {
88  CurrentPrcb->NextThread = NULL;
89  }
90 
91  //
92  // If a thread has already been selected to run on the current processor,
93  // then select that thread. Otherwise, attempt to select a thread from
94  // the current processor dispatcher ready queues.
95  //
96 
97  if ((NewThread = CurrentPrcb->NextThread) != NULL) {
98 
99  //
100  // Clear the next thread address, set the current thread address, and
101  // set the thread state to running.
102  //
103 
104  CurrentPrcb->NextThread = NULL;
105  CurrentPrcb->CurrentThread = NewThread;
106  NewThread->State = Running;
107 
108  } else {
109 
110  //
111  // Attempt to select a thread from the current processor dispatcher
112  // ready queues.
113  //
114 
115  NewThread = KiSelectReadyThread(0, CurrentPrcb);
116  if (NewThread != NULL) {
117  CurrentPrcb->NextThread = NULL;
118  CurrentPrcb->CurrentThread = NewThread;
119  NewThread->State = Running;
120 
121  } else {
122 
123  //
124  // Release the current PRCB lock and attempt to select a thread
125  // from any processor dispatcher ready queues.
126  //
127  // If this is a multinode system, then start with the processors
128  // on the same node. Otherwise, start with the current processor.
129  //
130 
131  KiReleasePrcbLock(CurrentPrcb);
132  Processor = CurrentPrcb->Number;
133  Index = Processor;
134  if (KeNumberNodes > 1) {
135  KeFindFirstSetLeftAffinity(CurrentPrcb->ParentNode->ProcessorMask,
136  (PULONG)&Index);
137  }
138 
139  Limit = KeNumberProcessors - 1;
140  Number = Limit;
141  do {
142  TargetPrcb = KiProcessorBlock[Index];
143  if (CurrentPrcb != TargetPrcb) {
144  if (TargetPrcb->ReadySummary != 0) {
145 
146  //
147  // Acquire the current and target PRCB locks.
148  //
149 
150  KiAcquireTwoPrcbLocks(CurrentPrcb, TargetPrcb);
151 
152  //
153  // If a new thread has not been selected to run on
154  // the current processor, then attempt to select a
155  // thread to run on the current processor.
156  //
157 
158  if ((NewThread = CurrentPrcb->NextThread) == NULL) {
159  if ((TargetPrcb->ReadySummary != 0) &&
160  (NewThread = KiFindReadyThread(Processor,
161  TargetPrcb)) != NULL) {
162 
163  //
164  // A new thread has been found to run on the
165  // current processor.
166  //
167 
168  NewThread->State = Running;
169  KiReleasePrcbLock(TargetPrcb);
170  CurrentPrcb->NextThread = NULL;
171  CurrentPrcb->CurrentThread = NewThread;
172 
173  //
174  // Clear idle on the current processor
175  // and update the idle SMT summary set to
176  // indicate the set is not idle.
177  //
178 
179  KiClearIdleSummary(AFFINITY_MASK(Processor));
180  KiClearSMTSummary(CurrentPrcb->MultiThreadProcessorSet);
181  goto ThreadFound;
182 
183  } else {
184  KiReleasePrcbLock(CurrentPrcb);
185  KiReleasePrcbLock(TargetPrcb);
186  }
187 
188  } else {
189 
190  //
191  // A thread has already been selected to run on
192  // the current processor. It is possible that
193  // the thread is the idle thread due to a state
194  // change that made a scheduled runable thread
195  // unrunable.
196  //
197  // N.B. If the idle thread is selected, then the
198  // current processor is idle. Otherwise,
199  // the current processor is not idle.
200  //
201 
202  if (NewThread == CurrentPrcb->IdleThread) {
203  CurrentPrcb->NextThread = NULL;
204  CurrentPrcb->IdleSchedule = FALSE;
205  KiReleasePrcbLock(CurrentPrcb);
206  KiReleasePrcbLock(TargetPrcb);
207  continue;
208 
209  } else {
210  NewThread->State = Running;
211  KiReleasePrcbLock(TargetPrcb);
212  CurrentPrcb->NextThread = NULL;
213  CurrentPrcb->CurrentThread = NewThread;
214  goto ThreadFound;
215  }
216  }
217  }
218  }
219 
220  Index -= 1;
221  if (Index < 0) {
222  Index = Limit;
223  }
224 
225  Number -= 1;
226  } while (Number >= 0);
227  }
228  }
229 
230  //
231  // If a new thread has been selected for execution, then release the
232  // PRCB lock and acquire the idle thread lock. Otherwise, lower IRQL
233  // to DISPATCH_LEVEL.
234  //
235 
236 ThreadFound:;
237  if (NewThread == NULL) {
238  KeLowerIrql(DISPATCH_LEVEL);
239 
240  } else {
241  KiSetContextSwapBusy(CurrentPrcb->IdleThread);
242  KiReleasePrcbLock(CurrentPrcb);
243  }
244 
245  return NewThread;
246 }
