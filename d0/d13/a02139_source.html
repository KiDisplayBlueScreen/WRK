11  zone.c
12 
13 Abstract:
14 
15  This module implements a simple zone buffer manager. The primary
16  consumer of this module is local LPC.
17 
18  The zone package provides a fast and efficient memory allocator for
19  fixed-size 64-bit aligned blocks of storage. The zone package does
20  not provide any serialization over access to the zone header and
21  associated free list and segment list. It is the responsibility of
22  the caller to provide any necessary serialization.
23 
24  The zone package views a zone as a set of fixed-size blocks of
25  storage. The block size of a zone is specified during zone
26  initialization. Storage is assigned to a zone during zone
27  initialization and when a zone is extended. In both of these cases,
28  a segment and length are specified.
29 
30  The zone package uses the first ZONE_SEGMENT_HEADER portion of the
31  segment for zone overhead. The remainder of the segment is carved
32  up into fixed-size blocks and each block is added to the free list
33  maintained in the zone header.
34 
35  As long as a block is on the free list, the first SINGLE_LIST_ENTRY
36  (32 bit) sized piece of the block is used as zone overhead. The
37  rest of the block is not used by the zone package and may be used by
38  applications to cache information. When a block is not on the free
39  list, its entire contents are available to the application.
40 
41 --*/

53 /*++
54 
55 Routine Description:
56 
57  This function initializes a zone header. Once successfully
58  initialized, blocks can be allocated and freed from the zone, and
59  the zone can be extended.
60 
61 Arguments:
62 
63  Zone - Supplies the address of a zone header to be initialized.
64 
65  BlockSize - Supplies the block size of the allocatable unit within
66  the zone. The size must be larger that the size of the
67  initial segment, and must be 64-bit aligned.
68 
69  InitialSegment - Supplies the address of a segment of storage. The
70  first ZONE_SEGMENT_HEADER-sized portion of the segment
71  is used by the zone allocator. The remainder of
72  the segment is carved up into fixed size
73  (BlockSize) blocks and is made available for
74  allocation and deallocation from the zone. The
75  address of the segment must be aligned on a 64-bit
76  boundary.
77 
78  InitialSegmentSize - Supplies the size in bytes of the InitialSegment.
79 
80 Return Value:
81 
82  STATUS_UNSUCCESSFUL - BlockSize or InitialSegment was not aligned on
83  64-bit boundaries, or BlockSize was larger than
84  the initial segment size.
85 
86  STATUS_SUCCESS - The zone was successfully initialized.
87 
88 --*/
89 
90 {
91  ULONG i;
92  PCH p;
93 
94  if ( (BlockSize & 7) || ((ULONG_PTR)InitialSegment & 7) ||
95  (BlockSize > InitialSegmentSize) ) {
96 #if DBG
97  DbgPrint( "EX: ExInitializeZone( %x, %x, %x, %x ) - Invalid parameters.\n",
98  Zone, BlockSize, InitialSegment, InitialSegmentSize
99  );
100  DbgBreakPoint();
101 #endif
102  return STATUS_INVALID_PARAMETER;
103  }
104 
105  Zone->BlockSize = BlockSize;
106 
107  Zone->SegmentList.Next = &((PZONE_SEGMENT_HEADER) InitialSegment)->SegmentList;
108  ((PZONE_SEGMENT_HEADER) InitialSegment)->SegmentList.Next = NULL;
109  ((PZONE_SEGMENT_HEADER) InitialSegment)->Reserved = NULL;
110 
111  Zone->FreeList.Next = NULL;
112 
113  p = (PCH)InitialSegment + sizeof(ZONE_SEGMENT_HEADER);
114 
115  for (i = sizeof(ZONE_SEGMENT_HEADER);
116  i <= InitialSegmentSize - BlockSize;
117  i += BlockSize
118  ) {
119  ((PSINGLE_LIST_ENTRY)p)->Next = Zone->FreeList.Next;
120  Zone->FreeList.Next = (PSINGLE_LIST_ENTRY)p;
121  p += BlockSize;
122  }
123  Zone->TotalSegmentSize = i;
124 
125  return STATUS_SUCCESS;
126 }

135 /*++
136 
137 Routine Description:
138 
139  This function extends a zone by adding another segment's worth of
140  blocks to the zone.
141 
142 Arguments:
143 
144  Zone - Supplies the address of a zone header to be extended.
145 
146  Segment - Supplies the address of a segment of storage. The first
147  ZONE_SEGMENT_HEADER-sized portion of the segment is used by the
148  zone allocator. The remainder of the segment is carved up
149  into fixed-size (BlockSize) blocks and is added to the
150  zone. The address of the segment must be aligned on a 64-
151  bit boundary.
152 
153  SegmentSize - Supplies the size in bytes of Segment.
154 
155 Return Value:
156 
157  STATUS_UNSUCCESSFUL - BlockSize or Segment was not aligned on
158  64-bit boundaries, or BlockSize was larger than
159  the segment size.
160 
161  STATUS_SUCCESS - The zone was successfully extended.
162 
163 --*/
164 
165 {
166  ULONG i;
167  PCH p;
168 
169  if ( ((ULONG_PTR)Segment & 7) ||
170  (SegmentSize & 7) ||
171  (Zone->BlockSize > SegmentSize) ) {
172  return STATUS_UNSUCCESSFUL;
173  }
174 
175  ((PZONE_SEGMENT_HEADER) Segment)->SegmentList.Next = Zone->SegmentList.Next;
176  Zone->SegmentList.Next = &((PZONE_SEGMENT_HEADER) Segment)->SegmentList;
177 
178  p = (PCH)Segment + sizeof(ZONE_SEGMENT_HEADER);
179 
180  for (i = sizeof(ZONE_SEGMENT_HEADER);
181  i <= SegmentSize - Zone->BlockSize;
182  i += Zone->BlockSize
183  ) {
184 
185  ((PSINGLE_LIST_ENTRY)p)->Next = Zone->FreeList.Next;
186  Zone->FreeList.Next = (PSINGLE_LIST_ENTRY)p;
187  p += Zone->BlockSize;
188  }
189  Zone->TotalSegmentSize += i;
190 
191  return STATUS_SUCCESS;
192 }
