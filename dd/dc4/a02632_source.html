84 /*++

86 Routine Description:
87 
88  This routine converts an NT status code to its DOS/OS 2 equivalent
89  and returns the translated value.
90 
91 Arguments:
92 
93  Status - Supplies the status value to convert.
94 
95 Return Value:
96 
97  The matching DOS/OS 2 error code.
98 
99 --*/
100 
101 {
102 
103  ULONG Offset;
104  ULONG Entry;
105  ULONG Index;
106 
107  //
108  // Convert any HRESULTs to their original form of a NTSTATUS or a
109  // WIN32 error
110  //
111 
112 
113  if (Status & 0x20000000) {
114 
115  //
116  // The customer bit is set so lets just pass the
117  // error code on thru
118  //
119 
120  return Status;
121 
122  }
123  else if ((Status & 0xffff0000) == 0x80070000) {
124 
125  //
126  // The status code was a win32 error already.
127  //
128 
129  return(Status & 0x0000ffff);
130  }
131  else if ((Status & 0xf0000000) == 0xd0000000) {
132 
133  //
134  // The status code is a HRESULT from NTSTATUS
135  //
136 
137  Status &= 0xcfffffff;
138  }
139 
140 
141  //
142  // Scan the run length table and compute the entry in the translation
143  // table that maps the specified status code to a DOS error code.
144  //
145 
146  Entry = 0;
147  Index = 0;
148  do {
149  if ((ULONG)Status >= RtlpRunTable[Entry + 1].BaseCode) {
150  Index += (RtlpRunTable[Entry].RunLength * RtlpRunTable[Entry].CodeSize);
151 
152  } else {
153  Offset = (ULONG)Status - RtlpRunTable[Entry].BaseCode;
154  if (Offset >= RtlpRunTable[Entry].RunLength) {
155  break;
156 
157  } else {
158  Index += (Offset * (ULONG)RtlpRunTable[Entry].CodeSize);
159  if (RtlpRunTable[Entry].CodeSize == 1) {
160  return (ULONG)RtlpStatusTable[Index];
161 
162  } else {
163  return (((ULONG)RtlpStatusTable[Index + 1] << 16) |
164  (ULONG)RtlpStatusTable[Index]);
165  }
166  }
167  }
168 
169  Entry += 1;
170  } while (Entry < (sizeof(RtlpRunTable) / sizeof(RUN_ENTRY)));
171 
172  //
173  // The translation to a DOS error code failed.
174  //
175  // The redirector maps unknown OS/2 error codes by ORing 0xC001 into
176  // the high 16 bits. Detect this and return the low 16 bits if true.
177  //
178 
179  if (((ULONG)Status >> 16) == 0xC001) {
180  return ((ULONG)Status & 0xFFFF);
181  }
182 
183  return ERROR_MR_MID_NOT_FOUND;
184 }
