38 ; Macro Description:
39 ;
40 ; This macro allocates exception record on stack, sets up exception
41 ; record using specified parameters and finally sets up arguments
42 ; and calls _KiDispatchException.
43 ;
44 ; Arguments:
45 ;
46 ; ExcepCode - Exception code to put into exception record
47 ; ExceptFlags - Exception flags to put into exception record
48 ; ExceptRecord - Associated exception record
49 ; ExceptAddress - Addr of instruction which the hardware exception occurs
50 ; NumParms - Number of additional parameters
51 ; ParameterList - the additional parameter list
52 ;
53 ; Return Value:
54 ;
55 ; None.
56 ;
57 ;--
58 
59 DISPATCH_EXCEPTION macro ExceptCode, ExceptFlags, ExceptRecord, ExceptAddress,\
60  NumParms, ParameterList
61  local de10, de20
62 
63 ; Set up exception record for raising exception
64 
65 ?i = 0
66  sub esp, ExceptionRecordSize + NumParms * 4
67  ; allocate exception record
68  mov dword ptr [esp]+ErExceptionCode, ExceptCode
69  ; set up exception code
70  mov dword ptr [esp]+ErExceptionFlags, ExceptFlags
71  ; set exception flags
72  mov dword ptr [esp]+ErExceptionRecord, ExceptRecord
73  ; set associated exception record
74  mov dword ptr [esp]+ErExceptionAddress, ExceptAddress
75  mov dword ptr [esp]+ErNumberParameters, NumParms
76  ; set number of parameters
77  IRP z, <ParameterList>
78  mov dword ptr [esp]+(ErExceptionInformation+?i*4), z
79 ?i = ?i + 1
80  ENDM
81 
82 ; set up arguments and call _KiDispatchException
83 
84  mov ecx, esp ; (ecx)->exception record
85 
86  test dword ptr [ebp]+TsEFlags,EFLAGS_V86_MASK
87  jz de10
88 
89  mov eax,0FFFFh
90  jmp de20
91 
92 de10: mov eax,[ebp]+TsSegCs
93 de20: and eax,MODE_MASK
94 
95 ; 1 - set first chance TRUE
96 ; eax - PreviousMode
97 ; ebp - trap frame addr
98 ; 0 - Null exception frame
99 ; ecx - exception record addr
100 
101 ; dispatchexception as appropriate
102  stdCall _KiDispatchException, <ecx, 0, ebp, eax, 1>
103  add esp,ExceptionRecordSize + NumParms*4
104 
105  ENDM

191 ;
192 ; ENTRY opc = opcode value
193 ; index = INDEX_ value corresponding to opcode
194 ; ?badindex = unspecified entry index
195 ; ?sopc = specific entry index

203  if ?sopc gt opc
