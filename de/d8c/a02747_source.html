197 void WmipEnableCollectionForNewGuid(
198  LPGUID Guid,
199  PBINSTANCESET InstanceSet
200  )
201 {
202  WNODE_HEADER Wnode;
203  PBGUIDENTRY GuidEntry;
204  ULONG Status;
205  BOOLEAN IsTraceLog;
206 
207  PAGED_CODE();
208 
209  GuidEntry = WmipFindGEByGuid(Guid, FALSE);
210 
211  if (GuidEntry != NULL)
212  {
213  memset(&Wnode, 0, sizeof(WNODE_HEADER));
214  memcpy(&Wnode.Guid, Guid, sizeof(GUID));
215  Wnode.BufferSize = sizeof(WNODE_HEADER);
216 
217  WmipEnterSMCritSection();
218  if ((GuidEntry->EventRefCount > 0) &&
219  ((InstanceSet->Flags & IS_ENABLE_EVENT) == 0))
220 
221  {
222  //
223  // Events were previously enabled for this guid, but not for this
224  // instance set so call data source for instance set to enable
225  // the events. First set the in progress flag and InstanceSet
226  // set flag to denote that events have been enabled for the
227  // instance set.
228  InstanceSet->Flags |= IS_ENABLE_EVENT;
229 
230  //
231  // If it is Tracelog, NewGuid notifications are piggybacked with
232  // Registration call return.
233  //
234  IsTraceLog = ((InstanceSet->Flags & IS_TRACED) == IS_TRACED) ? TRUE : FALSE;
235  if (IsTraceLog)
236  {
237  if (!(InstanceSet->DataSource->Flags & DS_KERNEL_MODE) )
238  {
239  if (GuidEntry != NULL)
240  {
241  WmipUnreferenceGE(GuidEntry);
242  }
243  WmipLeaveSMCritSection();
244  return;
245  }
246 
247  //
248  // For the Kernel Mode Trace Providers pass on the context
249  //
250  Wnode.HistoricalContext = GuidEntry->LoggerContext;
251  }
252 
253  GuidEntry->Flags |= GE_FLAG_NOTIFICATION_IN_PROGRESS;
254 
255  WmipLeaveSMCritSection();
256  WmipDeliverWnodeToDS(IRP_MN_ENABLE_EVENTS,
257  InstanceSet->DataSource,
258  &Wnode,
259  Wnode.BufferSize);
260  WmipEnterSMCritSection();
261 
262  //
263  // Now we need to check if events were disabled while the enable
264  // request was in progress. If so go do the work to actually
265  // disable them.
266  if (GuidEntry->EventRefCount == 0)
267  {
268  Status = WmipDoDisableRequest(GuidEntry,
269  TRUE,
270  IsTraceLog,
271  GuidEntry->LoggerContext,
272  GE_FLAG_NOTIFICATION_IN_PROGRESS);
273 
274  } else {
275  GuidEntry->Flags &= ~GE_FLAG_NOTIFICATION_IN_PROGRESS;
276  }
277  }
278 
279  //
280  // Now check to see if collection needs to be enabled for this guid
281  //
282  if ((GuidEntry->CollectRefCount > 0) &&
283  ((InstanceSet->Flags & IS_ENABLE_COLLECTION) == 0) &&
284  (InstanceSet->Flags & IS_EXPENSIVE) )
285 
286  {
287  //
288  // Collection was previously enabled for this guid, but not
289  // for this instance set so call data source for instance set
290  // to enable collection. First set the in progress flag and
291  // InstanceSet set flag to denote that collection has been enabled
292  // for the instance set.
293  //
294  GuidEntry->Flags |= GE_FLAG_COLLECTION_IN_PROGRESS;
295  InstanceSet->Flags |= IS_ENABLE_COLLECTION;
296 
297  WmipLeaveSMCritSection();
298  WmipDeliverWnodeToDS(IRP_MN_ENABLE_COLLECTION,
299  InstanceSet->DataSource,
300  &Wnode,
301  Wnode.BufferSize);
302  WmipEnterSMCritSection();
303 
304  //
305  // Now we need to check if events were disabled while the enable
306  // request was in progress. If so go do the work to actually
307  // disable them.
308  //
309  if (GuidEntry->CollectRefCount == 0)
310  {
311  Status = WmipDoDisableRequest(GuidEntry,
312  FALSE,
313  FALSE,
314  0,
315  GE_FLAG_COLLECTION_IN_PROGRESS);
316 
317  } else {
318  GuidEntry->Flags &= ~GE_FLAG_COLLECTION_IN_PROGRESS;
319 
320  //
321  // If there are any other threads that were waiting
322  // until all of the enable/disable work completed, we
323  // close the event handle to release them from their wait.
324  //
325  WmipReleaseCollectionEnabled(GuidEntry);
326  }
327  }
328  WmipUnreferenceGE(GuidEntry);
329  WmipLeaveSMCritSection();
330  } else {
331  WmipAssert(FALSE);
332  }

462 ULONG WmipDetermineInstanceBaseIndex(
463  LPGUID Guid,
464  PWCHAR BaseName,
465  ULONG InstanceCount
466  )
467 /*++
468 
469 Routine Description:
470 
471  Figure out the base index for the instance names specified by a base
472  instance name. We walk the list of instances sets for the guid and if
473  there is a match in the base instance name we set the base instance index
474  above that used by the previously registered instance set.
475 
476 Arguments:
477 
478  Guid points at guid for the instance names
479  BaseName points at the base name for the instances
480  InstanceCount is the count of instance names
481 
482 Return Value:
483 
484  Base index for instance name
485 
486 --*/
487 {
488  PBGUIDENTRY GuidEntry;
489  ULONG BaseIndex = 0;
490  PLIST_ENTRY InstanceSetList;
491  PBINSTANCESET InstanceSet;
492  ULONG LastBaseIndex;
493 
494  PAGED_CODE();
495 
496  UNREFERENCED_PARAMETER (InstanceCount);
497 
498  WmipEnterSMCritSection();
499 
500  GuidEntry = WmipFindGEByGuid(Guid, FALSE);
501  if (GuidEntry != NULL)
502  {
503  InstanceSetList = GuidEntry->ISHead.Flink;
504  while (InstanceSetList != &GuidEntry->ISHead)
505  {
506  InstanceSet = CONTAINING_RECORD(InstanceSetList,
507  INSTANCESET,
508  GuidISList);
509  if (InstanceSet->Flags & IS_INSTANCE_BASENAME)
510  {
511  if (wcscmp(BaseName, InstanceSet->IsBaseName->BaseName) == 0)
512  {
513  LastBaseIndex = InstanceSet->IsBaseName->BaseIndex + InstanceSet->Count;
514  if (BaseIndex <= LastBaseIndex)
515  {
516  BaseIndex = LastBaseIndex;
517  }
518  }
519  }
520  InstanceSetList = InstanceSetList->Flink;
521  }
522  WmipUnreferenceGE(GuidEntry);
523  }
524 
525  WmipLeaveSMCritSection();
526 
527  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Static instance name %ws has base index %x\n",
528  BaseName, BaseIndex));
529  return(BaseIndex);
530 }

532 NTSTATUS
533 WmipMangleInstanceName(
534  IN LPGUID Guid,
535  IN PWCHAR OriginalName,
536  IN SIZE_T MaxMangledNameLen,
537  OUT PWCHAR MangledName
538  )
539 /*++
540 
541 Routine Description:
542 
543  Copies a static instance name from the input buffer to the output
544  buffer, mangling it if the name collides with another name for the
545  same guid.
546 
547 Arguments:
548 
549  Guid - points at guid for the instance name
550 
551  OriginalName - points at the proposed, null-terminated instance name
552 
553  MaxMangledNameLen - has the maximum number of chars in mangled name buffer,
554  including the terminating NULL
555 
556  MangledName - points at buffer to return the null-terminated mangled name
557 
558 Return Value:
559 
560  Normal NTSTATUS code.
561 
562 Note:
563 
564  This algorithm is busted! It's supposed to try up to (26^6) mangled name
565  variations, but it actually only tries up to (26 * 6) variations.
566 
567  Note that, if we wanted to, we could actually go for (62^6) variations
568  if we used uppercase letters, lowercase letters and numbers.
569 
570 --*/
571 {
572  PBGUIDENTRY GuidEntry;
573  ULONG InstanceIndex;
574  PBINSTANCESET InstanceSet;
575  SIZE_T ManglePos;
576  WCHAR ManglingChar;
577  SIZE_T NameLength;
578  NTSTATUS Status;
579 
580  PAGED_CODE();
581 
582  WmipAssert(Guid != NULL);
583  WmipAssert(OriginalName != NULL);
584  WmipAssert(MangledName != NULL);
585 
586  GuidEntry = NULL;
587 
588  if (Guid == NULL ||
589  OriginalName == NULL ||
590  MangledName == NULL) {
591 
592  Status = STATUS_INVALID_PARAMETER;
593  goto Exit;
594  }
595 
596  NameLength = wcslen(OriginalName);
597 
598  WmipAssert(MaxMangledNameLen >= NameLength + 1);
599 
600  //
601  // To make prefast happy, we have to wcsncpy the string and manually
602  // terminate it.
603  //
604 
605  wcsncpy(MangledName, OriginalName, NameLength);
606  MangledName[NameLength] = UNICODE_NULL;
607 
608  Status = STATUS_SUCCESS;
609 
610  GuidEntry = WmipFindGEByGuid(Guid, FALSE);
611 
612  if (GuidEntry == NULL) {
613  goto Exit;
614  }
615 
616  //
617  // Set ManglePos to the last valid character in the name and set
618  // ManglingChar to 'Z' -- note that this will trigger the loop below
619  // to move ManglePos one position forward (to the first spot we actually
620  // want to mangle) and reset ManglingChar to 'A' to really start the
621  // mangling.
622  //
623 
624  ManglePos = NameLength - 1;
625  ManglingChar = L'Z';
626 
627  //
628  // Loop until we get a unique name.
629  //
630 
631  InstanceSet = WmipFindISinGEbyName(GuidEntry, MangledName, &InstanceIndex);
632 
633  while (InstanceSet != NULL) {
634 
635  WmipUnreferenceIS(InstanceSet);
636  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
637  DPFLTR_INFO_LEVEL,
638  "WMI: Need to mangle name %ws\n",
639  MangledName));
640 
641  if (ManglingChar == L'Z') {
642 
643  //
644  // Reset back to 'A' and move to the next character position to
645  // try and mangle.
646  //
647 
648  ManglingChar = L'A';
649  ++ManglePos;
650 
651  if (ManglePos == MaxMangledNameLen - 1) {
652 
653  //
654  // We're out of room, so fail.
655  //
656 
657  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
658  DPFLTR_INFO_LEVEL,
659  "WMI: Instance Name could not be mangled\n"));
660 
661  Status = STATUS_UNSUCCESSFUL;
662  goto Exit;
663  }
664 
665  //
666  // Make sure to re-terminate the string.
667  //
668 
669  MangledName[ManglePos + 1] = UNICODE_NULL;
670  } else {
671  ++ManglingChar;
672  }
673 
674  MangledName[ManglePos] = ManglingChar;
675 
676  InstanceSet = WmipFindISinGEbyName(GuidEntry,
677  MangledName,
678  &InstanceIndex) ;
679  }
680 
681 Exit:
682 
683  if (GuidEntry != NULL) {
684  WmipUnreferenceGE(GuidEntry);
685  }
686 
687  return Status;
688 }

701 Routine Description:

704  instance names as described in RegGuid.
705 
706 Arguments:
707 
708  RegGuid - pointer to registration information about this instance set

711 
712  BufferSize - total size of WmiRegInfo
713

717

723 
724 --*/
725 {
726  ULONG i;
727  ULONG InstanceCount;
728  PWCHAR InstanceName;
729  SIZE_T InstanceNameOffset;
730  PWCHAR InstanceNamePtr;
731  PBISBASENAME IsBaseName;
732  PBISSTATICNAMES IsStaticName;
733  SIZE_T MaxStaticInstanceNameLength;
734  SIZE_T NameLength;
735  SIZE_T SizeNeeded;
736  PWCHAR StaticInstanceNameBuffer;
737  PWCHAR StaticNames;
738  NTSTATUS Status;
739 
740  PAGED_CODE();
741 
742  WmipAssert(RegGuid != NULL);
743  WmipAssert(WmiRegInfo != NULL);
744  WmipAssert(BufferSize > 0);
745  WmipAssert(InstanceSet != NULL);
746 
747  UNREFERENCED_PARAMETER(MofImagePath);
748 
749  StaticInstanceNameBuffer = NULL;
750 
751  if (RegGuid == NULL ||
752  WmiRegInfo == NULL ||
753  BufferSize == 0 ||
754  InstanceSet == NULL) {
755 
756  Status = STATUS_INVALID_PARAMETER;
757  goto Exit;
758  }
759 
760  //
761  // Remember the count of instances for the guid in the DS.
762  //
763

765

768

772  // request comes in.
773  //
774 
775  InstanceSet->WADInstanceNameSize = 0;
776 
777  //
778  // Reset any flags that might be changed by a new REGGUID.
779  //
780 
781  InstanceSet->Flags &= ~(IS_EXPENSIVE |
782  IS_EVENT_ONLY |
783  IS_PDO_INSTANCENAME |
784  IS_INSTANCE_STATICNAMES |
785  IS_INSTANCE_BASENAME);
786

788  // Finish initializing the Instance Set flags.
789  //
790 
791  if (RegGuid->Flags & WMIREG_FLAG_EXPENSIVE) {
792  InstanceSet->Flags |= IS_EXPENSIVE;
793  }
794 
795  if (RegGuid->Flags & WMIREG_FLAG_TRACED_GUID) {
796 
797  //
798  // This guid is not queryable, but is used for sending trace
799  // events. We mark the InstanceSet as special.

995  goto Exit;
996  }

999  ++InstanceNamePtr;

1002  ((NameLength + 1) * sizeof(WCHAR)));
