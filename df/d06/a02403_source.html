33  IN PKTHREAD Thread,

40 /*++
41 
42 Routine Description:
43 
44  This function initializes the machine dependent context of a thread object.
45 
46  N.B. This function does not check the accessibility of the context record.
47  It is assumed the the caller of this routine is either prepared to
48  handle access violations or has probed and copied the context record
49  as appropriate.
50 
51 Arguments:
52 
53  Thread - Supplies a pointer to a dispatcher object of type thread.
54 
55  SystemRoutine - Supplies a pointer to the system function that is to be
56  called when the thread is first scheduled for execution.
57 
58  StartRoutine - Supplies an optional pointer to a function that is to be
59  called after the system has finished initializing the thread. This
60  parameter is specified if the thread is a system thread and will
61  execute totally in kernel mode.
62 
63  StartContext - Supplies an optional pointer to an arbitrary data structure
64  which will be passed to the StartRoutine as a parameter. This
65  parameter is specified if the thread is a system thread and will
66  execute totally in kernel mode.
67 
68  ContextFrame - Supplies an optional pointer a context frame which contains
69  the initial user mode state of the thread. This parameter is specified
70  if the thread is a user thread and will execute in user mode. If this
71  parameter is not specified, then the Teb parameter is ignored.
72 
73 Return Value:
74 
75  None.
76 
77 --*/
78 
79 {
80 
81  PFX_SAVE_AREA NpxFrame;
82  PKSWITCHFRAME SwitchFrame;
83  PKTRAP_FRAME TrFrame;
84  PULONG PSystemRoutine;
85  PULONG PStartRoutine;
86  PULONG PStartContext;
87  PULONG PUserContextFlag;
88  ULONG ContextFlags;
89  CONTEXT Context2;
90  PCONTEXT ContextFrame2 = NULL;
91  PFXSAVE_FORMAT PFxSaveArea;
92 
93  //
94  // If a context frame is specified, then initialize a trap frame and
95  // and an exception frame with the specified user mode context.
96  //
97 
98  if (ARGUMENT_PRESENT(ContextFrame)) {
99 
100  RtlCopyMemory(&Context2, ContextFrame, sizeof(CONTEXT));
101  ContextFrame2 = &Context2;
102  ContextFlags = CONTEXT_CONTROL;
103 
104  //
105  // The 80387 save area is at the very base of the kernel stack.
106  //
107 
108  NpxFrame = (PFX_SAVE_AREA)(((ULONG)(Thread->InitialStack) -
109  sizeof(FX_SAVE_AREA)));
110 
111  TrFrame = (PKTRAP_FRAME)(((ULONG)NpxFrame - KTRAP_FRAME_LENGTH));
112 
113  //
114  // Zero out the trap frame and save area
115  //
116 
117  RtlZeroMemory(TrFrame, KTRAP_FRAME_LENGTH + sizeof(FX_SAVE_AREA));
118 
119  //
120  // Load up an initial NPX state.
121  //
122 
123  if (KeI386FxsrPresent == TRUE) {
124  PFxSaveArea = (PFXSAVE_FORMAT)ContextFrame2->ExtendedRegisters;
125 
126  PFxSaveArea->ControlWord = 0x27f; // like fpinit but 64bit mode
127  PFxSaveArea->StatusWord = 0;
128  PFxSaveArea->TagWord = 0;
129  PFxSaveArea->ErrorOffset = 0;
130  PFxSaveArea->ErrorSelector = 0;
131  PFxSaveArea->DataOffset = 0;
132  PFxSaveArea->DataSelector = 0;
133  PFxSaveArea->MXCsr = 0x1f80; // mask all the exceptions
134  } else {
135  ContextFrame2->FloatSave.ControlWord = 0x27f; // like fpinit but 64bit mode
136  ContextFrame2->FloatSave.StatusWord = 0;
137  ContextFrame2->FloatSave.TagWord = 0xffff;
138  ContextFrame2->FloatSave.ErrorOffset = 0;
139  ContextFrame2->FloatSave.ErrorSelector = 0;
140  ContextFrame2->FloatSave.DataOffset = 0;
141  ContextFrame2->FloatSave.DataSelector = 0;
142  }
143 
144 
145  if (KeI386NpxPresent) {
146  ContextFrame2->FloatSave.Cr0NpxState = 0;
147  NpxFrame->Cr0NpxState = 0;
148  NpxFrame->NpxSavedCpu = 0;
149  if (KeI386FxsrPresent == TRUE) {
150  ContextFlags |= CONTEXT_EXTENDED_REGISTERS;
151  } else {
152  ContextFlags |= CONTEXT_FLOATING_POINT;
153  }
154 
155  //
156  // Threads NPX state is not in the coprocessor.
157  //
158 
159  Thread->NpxState = NPX_STATE_NOT_LOADED;
160  Thread->Header.NpxIrql = PASSIVE_LEVEL;
161 
162  } else {
163  NpxFrame->Cr0NpxState = CR0_EM;
164 
165  //
166  // Threads NPX state is not in the coprocessor.
167  // In the emulator case, do not set the CR0_EM bit as their
168  // emulators may not want exceptions on FWAIT instructions.
169  //
170 
171  Thread->NpxState = NPX_STATE_NOT_LOADED & ~CR0_MP;
172  }
173 
174  //
175  // Force debug registers off. They won't work anyway from an
176  // initial frame, debuggers must set a hard breakpoint in the target
177  //
178 
179  ContextFrame2->ContextFlags &= ~CONTEXT_DEBUG_REGISTERS;
180 
181  //
182  // Space for arguments to KiThreadStartup. Order is important,
183  // Since args are passed on stack through KiThreadStartup to
184  // PStartRoutine with PStartContext as an argument.
185  //
186 
187  PUserContextFlag = (PULONG)TrFrame - 1;
188  PStartContext = PUserContextFlag - 1;
189  PStartRoutine = PStartContext - 1;
190  PSystemRoutine = PStartRoutine - 1;
191 
192  SwitchFrame = (PKSWITCHFRAME)((PUCHAR)PSystemRoutine -
193  sizeof(KSWITCHFRAME));
194 
195  //
196  // Copy information from the specified context frame to the trap and
197  // exception frames.
198  //
199 
200  KeContextToKframes(TrFrame, NULL, ContextFrame2,
201  ContextFrame2->ContextFlags | ContextFlags,
202  UserMode);
203 
204  TrFrame->HardwareSegSs |= RPL_MASK;
205  TrFrame->SegDs |= RPL_MASK;
206  TrFrame->SegEs |= RPL_MASK;
207  TrFrame->Dr7 = 0;
208 
209 #if DBG
210 
211  TrFrame->DbgArgMark = 0xBADB0D00;
212 
213 #endif
214 
215  //
216  // Tell KiThreadStartup that a user context is present.
217  //
218 
219  *PUserContextFlag = 1;
220 
221 
222  //
223  // Initialize the kernel mode ExceptionList pointer
224  //
225 
226  TrFrame->ExceptionList = EXCEPTION_CHAIN_END;
227 
228  //
229  // Initialize the saved previous processor mode.
230  //
231 
232  TrFrame->PreviousPreviousMode = UserMode;
233 
234  //
235  // Set the previous mode in thread object to user.
236  //
237 
238  Thread->PreviousMode = UserMode;
239 
240 
241  } else {
242 
243  //
244  // Dummy floating save area. Kernel threads don't have or use
245  // the floating point - the dummy save area is make the stacks
246  // consistent.
247  //
248 
249  NpxFrame = (PFX_SAVE_AREA)(((ULONG)(Thread->InitialStack) -
250  sizeof(FX_SAVE_AREA)));
251 
252  //
253  // Load up an initial NPX state.
254  //
255  RtlZeroMemory((PVOID)NpxFrame, sizeof(FX_SAVE_AREA));
256 
257  if (KeI386FxsrPresent == TRUE) {
258  NpxFrame->U.FxArea.ControlWord = 0x27f;//like fpinit but 64bit mode
259  NpxFrame->U.FxArea.MXCsr = 0x1f80;// mask all the exceptions
260 
261  } else {
262  NpxFrame->U.FnArea.ControlWord = 0x27f;//like fpinit but 64bit mode
263  NpxFrame->U.FnArea.TagWord = 0xffff;
264  }
265 
266  //
267  // Threads NPX state is not in the coprocessor.
268  //
269 
270  Thread->NpxState = NPX_STATE_NOT_LOADED;
271 
272  //
273  // Space for arguments to KiThreadStartup.
274  // Order of fields in the switchframe is important,
275  // Since args are passed on stack through KiThreadStartup to
276  // PStartRoutine with PStartContext as an argument.
277  //
278 
279  PUserContextFlag = (PULONG)((ULONG)NpxFrame) - 1;
280 
281  PStartContext = PUserContextFlag - 1;
282  PStartRoutine = PStartContext - 1;
283  PSystemRoutine = PStartRoutine - 1;
284 
285  SwitchFrame = (PKSWITCHFRAME)((PUCHAR)PSystemRoutine -
286  sizeof(KSWITCHFRAME));
287 
288  //
289  // Tell KiThreadStartup that a user context is NOT present.
290  //
291 
292  *PUserContextFlag = 0;
293 
294 
295  //
296  // Set the previous mode in thread object to kernel.
297  //
298 
299  Thread->PreviousMode = KernelMode;
300  }
301 
302  //
303  // Set up thread start parameters.
304  // (UserContextFlag set above)
305  //
306 
307  *PStartContext = (ULONG)StartContext;
308  *PStartRoutine = (ULONG)StartRoutine;
309  *PSystemRoutine = (ULONG)SystemRoutine;
310 
311  //
312  // Set up switch frame. Assume the thread doesn't use the 80387;
313  // if it ever does (and there is one), these flags will get reset.
314  // Each thread starts with these same flags set, regardless of
315  // whether the hardware exists or not.
316  //
317 
318  SwitchFrame->RetAddr = (ULONG)KiThreadStartup;
319  SwitchFrame->ApcBypassDisable = TRUE;
320  SwitchFrame->ExceptionList = (ULONG)(EXCEPTION_CHAIN_END);
321 
322 #if DBG
323 
324  //
325  // On checked builds add a check field so context swap can break
326  // early on bad context swaps (corrupted stacks for example).
327  // We place this below the stack pointer so the kernel debugger
328  // doesn't need knowledge of this.
329  //
330 
331  ((PULONG)SwitchFrame)[-1] = (ULONG)(ULONG_PTR)Thread;
332 
333 #endif
334 
335  //
336  // Set the initial kernel stack pointer.
337  //
338 
339  Thread->KernelStack = (PVOID)SwitchFrame;
340  return;
341 }
