73 CcMapData (
74  __in PFILE_OBJECT FileObject,
75  __in PLARGE_INTEGER FileOffset,
76  __in ULONG Length,
77  __in ULONG Flags,
78  __out PVOID *Bcb,
79  __deref_out_bcount(Length) PVOID *Buffer

82 /*++
83 
84 Routine Description:
85 
86  This routine attempts to map the specified file data in the cache.
87  A pointer is returned to the desired data in the cache.
88 
89  If the caller does not want to block on this call, then
90  Wait should be supplied as FALSE. If Wait was supplied as FALSE and
91  it is currently impossible to supply the requested data without
92  blocking, then this routine will return FALSE. However, if the
93  data is immediately accessible in the cache and no blocking is
94  required, this routine returns TRUE with a pointer to the data.
95 
96  Note that a call to this routine with Wait supplied as TRUE is
97  considerably faster than a call with Wait supplies as FALSE, because
98  in the Wait TRUE case we only have to make sure the data is mapped
99  in order to return.
100 
101  It is illegal to modify data that is only mapped, and can in fact lead
102  to serious problems. It is impossible to check for this in all cases,
103  however CcSetDirtyPinnedData may implement some Assertions to check for
104  this. If the caller wishes to modify data that it has only mapped, then
105  it must *first* call CcPinMappedData.
106 
107  In any case, the caller MUST subsequently call CcUnpinData.
108  Naturally if CcPinRead or CcPreparePinWrite were called multiple
109  times for the same data, CcUnpinData must be called the same number
110  of times.
111 
112  The returned Buffer pointer is valid until the data is unpinned, at
113  which point it is invalid to use the pointer further. This buffer pointer
114  will remain valid if CcPinMappedData is called.
115 
116  Note that under some circumstances (like Wait supplied as FALSE or more
117  than a page is requested), this routine may actually pin the data, however
118  it is not necessary, and in fact not correct, for the caller to be concerned
119  about this.
120 
121 Arguments:
122 
123  FileObject - Pointer to the file object for a file which was
124  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
125  which CcInitializeCacheMap was called by the file system.
126 
127  FileOffset - Byte offset in file for desired data.
128 
129  Length - Length of desired data in bytes.
130 
131  Wait - FALSE if caller may not block, TRUE otherwise (see description
132  above)
133 
134  Bcb - On the first call this returns a pointer to a Bcb
135  parameter which must be supplied as input on all subsequent
136  calls, for this buffer
137 
138  Buffer - Returns pointer to desired data, valid until the buffer is
139  unpinned or freed. This pointer will remain valid if CcPinMappedData
140  is called.
141 
142 Return Value:
143 
144  FALSE - if Wait was supplied as FALSE and the data was not delivered
145 
146  TRUE - if the data is being delivered
147 
148 --*/
149 
150 {
151  ULONG SavedState;
152  volatile UCHAR ch;
153  PVOID TempBcb;
154  PVOID BaseAddress;
155  ULONG PageCount = ADDRESS_AND_SIZE_TO_SPAN_PAGES((ULongToPtr(FileOffset->LowPart)), Length);
156  PETHREAD Thread = PsGetCurrentThread();
157  BOOLEAN ReturnStatus;
158 
159  DebugTrace(+1, me, "CcMapData\n", 0 );
160 
161  MmSavePageFaultReadAhead( Thread, &SavedState );
162 
163  ReturnStatus = CcMapDataCommon( FileObject,
164  FileOffset,
165  Length,
166  Flags,
167  &TempBcb,
168  Buffer );
169 
170  if (!ReturnStatus) {
171 
172  //
173  // No need to revert the CcMissCounter in this case because we can
174  // only get here if Wait was FALSE and the CcMissCounter isn't set
175  // by CcMapDataCommon if Wait is FALSE.
176  //
177 
178  return FALSE;
179  }
180 
181  //
182  // Caller specifically requested he doesn't want data to be faulted in.
183  //
184 
185  if (!FlagOn( Flags, MAP_NO_READ )) {
186 
187  //
188  // Now let's just sit here and take the miss(es) like a man (and count them).
189  //
190 
191  try {
192 
193  //
194  // Loop to touch each page
195  //
196 
197  BaseAddress = *Buffer;
198 
199  while (PageCount != 0) {
200 
201  MmSetPageFaultReadAhead( Thread, PageCount - 1 );
202 
203  ch = *((volatile UCHAR *)(BaseAddress));
204 
205  BaseAddress = (PCHAR) BaseAddress + PAGE_SIZE;
206  PageCount -= 1;
207  }
208 
209  } finally {
210 
211  MmResetPageFaultReadAhead( Thread, SavedState );
212 
213  if (AbnormalTermination() && (TempBcb != NULL)) {
214  CcUnpinFileData( (PBCB)TempBcb, TRUE, UNPIN );
215  }
216  }
217  }
218 
219  //
220  // CcMapDataCommon set the appropriate miss counter, but we need to reset
221  // it at this point.
222  //
223 
224  CcMissCounter = &CcThrowAway;
225 
226  //
227  // Increment the pointer as a reminder that it is read only, and
228  // return it. We pend this until now to avoid raising with a valid
229  // Bcb into caller's contexts.
230  //
231 
232  *(PCHAR *)&TempBcb += 1;
233  *Bcb = TempBcb;
234 
235  DebugTrace(-1, me, "CcMapData -> TRUE\n", 0 );
236 
237  return TRUE;
238 }

252 
253 Routine Description:
254 
255  This routine attempts to map the specified file data in the cache.
256  A pointer is returned to the desired data in the cache.
257 
258  If the caller does not want to block on this call, then
259  Wait should be supplied as FALSE. If Wait was supplied as FALSE and
260  it is currently impossible to supply the requested data without
261  blocking, then this routine will return FALSE. However, if the
262  data is immediately accessible in the cache and no blocking is

264

266  considerably faster than a call with Wait supplies as FALSE, because
267  in the Wait TRUE case we only have to make sure the data is mapped
268  in order to return.
269 
270  If data is only mapped, Cc does not setup the structures to track the
271  pages as dirty. If the caller plans to dirty the data, it must do one
272  of the following:
273  * Call CcPinMappedData - Cc will then setup the structures to track that
274  the buffer is dirty. Then the caller can dirty the buffer and call
275  CcSetDirtyPinnedData to tell Cc about it.
276  * Track the dirty data itself. If the caller chooses to do this, he/she
277  is responsible for setting the address range modified after it is
278  written, but before trying to flush the data.
279

281  Naturally if CcPinRead or CcPreparePinWrite were called multiple
282  times for the same data, CcUnpinData must be called the same number
283  of times.
284

286  which point it is invalid to use the pointer further. This buffer pointer
287  will remain valid if CcPinMappedData is called.
288 
289  Note that under some circumstances (like Wait supplied as FALSE or more
290  than a page is requested), this routine may actually pin the data, however
291  it is not necessary, and in fact not correct, for the caller to be concerned
292  about this.
293 
294 Arguments:
295 
296  FileObject - Pointer to the file object for a file which was
297  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
298  which CcInitializeCacheMap was called by the file system.
299 
300  FileOffset - Byte offset in file for desired data.
301 
302  Length - Length of desired data in bytes.
303 
304  Wait - FALSE if caller may not block, TRUE otherwise (see description
305  above)
306 
307  Bcb - On the first call this returns a pointer to a Bcb
308  parameter which must be supplied as input on all subsequent
309  calls, for this buffer
310 
311  Buffer - Returns pointer to desired data, valid until the buffer is
312  unpinned or freed. This pointer will remain valid if CcPinMappedData
313  is called.
314 
315 Return Value:
316 
317  FALSE - if Wait was supplied as FALSE and the data was not delivered
318 
319  TRUE - if the data is being delivered
320 
321 --*/

324  PSHARED_CACHE_MAP SharedCacheMap;
325  LARGE_INTEGER BeyondLastByte;
326  ULONG ReceivedLength;
327  PVOID TempBcb;
328  PETHREAD Thread = PsGetCurrentThread();
329 
330  DebugTrace(+1, me, "CcMapDataCommon\n", 0 );
331 
332  //
333  // Increment performance counters
334  //
335

339

385

387

389 
390  }

394  // caller to mark it as such since the caller will need to do some
395  // addition operations on this pointer before returning to the request
396  // originator.
397  //
398

401  DebugTrace(-1, me, "CcMapDataCommon -> TRUE\n", 0 );

407 BOOLEAN
408 CcPinMappedData (
409  __in PFILE_OBJECT FileObject,
410  __in PLARGE_INTEGER FileOffset,
411  __in ULONG Length,
412  __in ULONG Flags,
413  __inout PVOID *Bcb
414  )
415 
416 /*++
417 
418 Routine Description:
419 
420  This routine attempts to pin data that was previously only mapped.
421  If the routine determines that in fact it was necessary to actually
422  pin the data when CcMapData was called, then this routine does not
423  have to do anything.
424 
425  If the caller does not want to block on this call, then
426  Wait should be supplied as FALSE. If Wait was supplied as FALSE and
427  it is currently impossible to supply the requested data without
428  blocking, then this routine will return FALSE. However, if the
429  data is immediately accessible in the cache and no blocking is
430  required, this routine returns TRUE with a pointer to the data.
431 
432  If the data is not returned in the first call, the caller
433  may request the data later with Wait = TRUE. It is not required
434  that the caller request the data later.
435 
436  If the caller subsequently modifies the data, it should call
437  CcSetDirtyPinnedData.
438 
439  In any case, the caller MUST subsequently call CcUnpinData.
440  Naturally if CcPinRead or CcPreparePinWrite were called multiple
441  times for the same data, CcUnpinData must be called the same number
442  of times.
443 
444  Note there are no performance counters in this routine, as the misses
445  will almost always occur on the map above, and there will seldom be a
446  miss on this conversion.
447 
448 Arguments:
449 
450  FileObject - Pointer to the file object for a file which was
451  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
452  which CcInitializeCacheMap was called by the file system.
453 
454  FileOffset - Byte offset in file for desired data.
455 
456  Length - Length of desired data in bytes.
457 
458  Flags - (PIN_WAIT, PIN_EXCLUSIVE, PIN_NO_READ, etc. as defined in cache.h)
459  If the caller specifies PIN_NO_READ and PIN_EXCLUSIVE, then he must
460  guarantee that no one else will be attempting to map the view, if he
461  wants to guarantee that the Bcb is not mapped (view may be purged).
462  If the caller specifies PIN_NO_READ without PIN_EXCLUSIVE, the data
463  may or may not be mapped in the return Bcb.
464 
465  Bcb - On the first call this returns a pointer to a Bcb
466  parameter which must be supplied as input on all subsequent
467  calls, for this buffer
468 
469 Return Value:
470 
471  FALSE - if Wait was not set and the data was not delivered
472 
473  TRUE - if the data is being delivered
474 
475 --*/
476 
477 {
478  PVOID Buffer;
479  LARGE_INTEGER BeyondLastByte;
480  PSHARED_CACHE_MAP SharedCacheMap;
481  LARGE_INTEGER LocalFileOffset = *FileOffset;
482  POBCB MyBcb = NULL;
483  PBCB *CurrentBcbPtr = (PBCB *)&MyBcb;
484  BOOLEAN Result = FALSE;
485 
486  DebugTrace(+1, me, "CcPinMappedData\n", 0 );
487 
488  //
489  // If the Bcb is no longer ReadOnly, then just return.
490  //
491 
492  if ((*(PULONG)Bcb & 1) == 0) {
493  return TRUE;
494  }
495 
496  //
497  // Remove the Read Only flag
498  //
499 
500  *(PCHAR *)Bcb -= 1;
501 
502  //
503  // Get pointer to SharedCacheMap.
504  //
505 
506  SharedCacheMap = FileObject->SectionObjectPointer->SharedCacheMap;
507 
508  //
509  // We only count the calls to this routine, since they are almost guaranteed
510  // to be hits.
511  //
512 
513  CcPinMappedDataCount += 1;
514 
515  //
516  // Guarantee we will put the flag back if required.
517  //
518 
519  try {
520 
521  if (((PBCB)*Bcb)->NodeTypeCode != CACHE_NTC_BCB) {
522 
523  //
524  // Form loop to handle occasional overlapped Bcb case.
525  //
526 
527  do {
528 
529  //
530  // If we have already been through the loop, then adjust
531  // our file offset and length from the last time.
532  //
533 
534  if (MyBcb != NULL) {
535 
536  //
537  // If this is the second time through the loop, then it is time
538  // to handle the overlap case and allocate an OBCB.
539  //
540 
541  if (CurrentBcbPtr == (PBCB *)&MyBcb) {
542 
543  MyBcb = CcAllocateObcb( FileOffset, Length, (PBCB)MyBcb );
544 
545  //
546  // Set CurrentBcbPtr to point at the first entry in
547  // the vector (which is already filled in), before
548  // advancing it below.
549  //
550 
551  CurrentBcbPtr = &MyBcb->Bcbs[0];
552  }
553 
554  Length -= (ULONG)(BeyondLastByte.QuadPart - LocalFileOffset.QuadPart);
555  LocalFileOffset.QuadPart = BeyondLastByte.QuadPart;
556  CurrentBcbPtr += 1;
557  }
558 
559  //
560  // Call local routine to Map or Access the file data. If we cannot map
561  // the data because of a Wait condition, return FALSE.
562  //
563 
564  if (!CcPinFileData( FileObject,
565  &LocalFileOffset,
566  Length,
567  (BOOLEAN)!FlagOn(SharedCacheMap->Flags, MODIFIED_WRITE_DISABLED),
568  FALSE,
569  Flags,
570  CurrentBcbPtr,
571  &Buffer,
572  &BeyondLastByte )) {
573 
574  try_return( Result = FALSE );
575  }
576 
577  //
578  // Continue looping if we did not get everything.
579  //
580 
581  } while((BeyondLastByte.QuadPart - LocalFileOffset.QuadPart) < Length);
582 
583  //
584  // Free the Vacb before going on.
585  //
586 
587  CcFreeVirtualAddress( (PVACB)*Bcb );
588 
589  *Bcb = MyBcb;
590 
591  //
592  // Debug routines used to insert and remove Bcbs from the global list
593  //
594 
595  }
596 
597  //
598  // If he really has a Bcb, all we have to do is acquire it shared since he is
599  // no longer ReadOnly.
600  //
601 
602  else {
603 
604  if (!ExAcquireSharedStarveExclusive( &((PBCB)*Bcb)->Resource, BooleanFlagOn(Flags, PIN_WAIT))) {
605 
606  try_return( Result = FALSE );
607  }
608  }
609 
610  Result = TRUE;
611 
612  try_exit: NOTHING;
613  }
614  finally {
615 
616  if (!Result) {
617 
618  //
619  // Put the Read Only flag back
620  //
621 
622  *(PCHAR *)Bcb += 1;
623 
624  //
625  // We may have gotten partway through
626  //
627 
628  if (MyBcb != NULL) {
629  CcUnpinData( MyBcb );
630  }
631  }
632 
633  DebugTrace(-1, me, "CcPinMappedData -> %02lx\n", Result );
634  }
635  return Result;
636 }
